//判断链表是否有环
bool isloop(pNode phead)
{
	pNode fast = phead;
	pNode slow = phead;
	while (fast != NULL && fast->next != NULL)
	{
		fast = fast->next->next;
		slow = slow->next;
		if (fast == slow)
		{
			printf("链表有环\n");
			return true;
		}
		else
		{
		    printf("链表无环\n");
		    return false;
		}

	}
}

int GetloopLen(pNode phead)
{

	pNode fast = phead;
	pNode slow = phead;
                pNode mark=phead;
	fast = fast->next->next;
	slow = slow->next;
	if(fast==slow)
		mark=slow;
	do
	{
		len++;
		mark=mark->next;
	}while(slow==mark)
	return len;
}

//判断环的起点
pNode findloopentrance(pNode phead)
{
	pNode fast = phead;
	pNode slow = phead;
	while (fast != NULL && fast->next != NULL)
	{
		fast = fast->next->next;
		slow = slow->next;
		if (fast == slow)
			break;
	}
	slow = phead;
	while (slow != fast)
	{
		fast = fast->next;
		slow = slow->next;
	}
	return slow;
}
